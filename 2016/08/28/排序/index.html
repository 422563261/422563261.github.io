  • 冒泡排序：最简单，也最慢，貌似长度小于7最优
  • 快速排序：这是一个非常快的排序方式，V8的sort方法就使用快速排序和插入排序的结合
  • 插入排序： 比冒泡快，比快速排序和希尔排序慢，较小数据有优势
  • 希尔排序：在非chrome下数组长度小于1000，希尔排序比快速更快

//冒泡排序
function bubble (array) {
    let tmp,
        i,
        j;
    for (i = 0; i < array.length; i++) {
        for (var j = i+1; j < array.length; j++) {
            if (array[i] > array[j]) {
                tmp = array[j];
                array[j] = array[i];
                array[i] = tmp;
            }
        }
    }
    return array;  
}

//快速排序
function quick (array) {
    let i = 0,
        j = array.length - 1,
        Sort = function(i, j) {
            // 结束条件
            if (i == j) {
                return
            };

            let key = array[i],
                stepi = i, // 记录开始位置
                stepj = j; // 记录结束位置
            while (j > i) {
                // j <<-------------- 向前查找
                if (array[j] >= key) {
                    j--;
                } else {
                    array[i] = array[j]
                    //i++ ------------>>向后查找
                    while (j > ++i) {
                        if (array[i] > key) {
                            array[j] = array[i];
                            break;
                        }
                    }
                }
            }

            // 如果第一个取出的 key 是最小的数
            if (stepi == i) {
                Sort(++i, stepj);
                return;
            }

            // 最后一个空位留给 key
            array[i] = key;

            // 递归
            Sort(stepi, i);
            Sort(j, stepj);
        }
          Sort(i, j);
          return array;
}

//插入排序
function insert (array) {
    var i = 1,
    j, step, key, len = array.length;
    for (; i < len; i++) {
        step = j = i;
        key = array[j];//key代表新加入的元素
        while (--j > -1) {
        //从整理好的数组右端开始检测
            if (array[j] > key) {
                array[j + 1] = array[j];
                //若比较的值大于key，将取代key的位置
            } else {
                break;
            }
        }
        array[j + 1] = key;
        //将key放入合适的位置
    }
    return array;
}

            /*j < stepLen && */
            step * j + i < len; j++) { //依次循环每列的每行
                tem = f = step * j + i;
                key = array[f];

                        array[tem + step] = array[tem];
