2016-03-18 
<!DOCTYPE html>
<html lang="en">
<head>
<meta charset="UTF-8">
<title>Document</title>
<style>
    *{padding:0;margin:0;}
    #ball{
        width: 20px;
        height: 20px;
        border-radius:50%;
        border: 1px solid black;
        background: black;
        position: absolute;
    }
</style>
</head>
<body>
    <div id="ball" style="top: 0px;left: 0px"></div>
</body>
<script>
    var ball = document.getElementById('ball');
    var    speedno = 0;
    var xspeed = 0.8;
    var yspeed = 0;
    function animate(move) {
        lastdown = parseFloat(ball.style.top) + move;
        flashdown();
    }
    animate(320,0,0.8);

    function flashdown(){
        yspeed = yspeed + 0.015;
        console.log(xspeed);
        ball.style.top = parseFloat(ball.style.top) + yspeed + 'px';
        ball.style.left = parseFloat(ball.style.left) + xspeed + 'px';
        //console.log(ball.style.left)
        if (parseFloat(ball.style.top) < lastdown) {
            setTimeout(flashdown, 1);
        }
        else{
            if (yspeed == 0.015) {
                var stop = setInterval(function(){
                    xspeed = xspeed - 0.007;
                    ball.style.left = parseFloat(ball.style.left) + xspeed + 'px';
                    console.log(xspeed)
                    if (xspeed <=0 ) {
                        clearInterval(stop);
                    }
                },1);

            }
            else{
                speedno = -yspeed + 0.5;
                flashup();
            }

        }    
    }
    function flashup(){
        speedno = speedno + 0.015;
        ball.style.top = parseFloat(ball.style.top) + speedno + 'px';
        ball.style.left = parseFloat(ball.style.left) + xspeed + 'px';
        if (speedno < 0) {
            setTimeout(flashup,1);
        }
        else{
            yspeed = 0;
            flashdown();
        }
    }
</script>
</html>
